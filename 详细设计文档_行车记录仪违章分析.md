### 5.2 帧提取策略

```kotlin
class AdaptiveFrameExtractor {
    companion object {
        const val NORMAL_FPS = 15          // 正常检测帧率
        const val DETAIL_FPS = 30         // 违章前后提升帧率
        const val PRE_BUFFER_MS = 5100    // 预缓冲5.1秒
        const val POST_BUFFER_MS = 5100   // 后缓冲5.1秒
    }
    
    fun calculateSampleInterval(frameRate: Float, isNearViolation: Boolean): Long {
        val targetFps = if (isNearViolation) DETAIL_FPS else NORMAL_FPS
        return (1000 / targetFps).toLong()
    }
}
```

### 5.3 截图采集策略

```kotlin
suspend fun captureViolationScreenshots(
    videoPath: String,
    violationTimeMs: Long,
    outputDir: String,
    vehicle: Vehicle
): Result<List<Screenshot>> {
    val timestamps = listOf(
        violationTimeMs - 5000,  // T-5s
        violationTimeMs,         // T0
        violationTimeMs + 5000   // T+5s
    )
    
    return withContext(Dispatchers.IO) {
        timestamps.mapIndexed { index, timestamp ->
            val frame = frameExtractor.extractFrame(timestamp)
                ?: return@withContext Result.failure(...)
            
            val screenshot = saveScreenshot(
                frame = frame,
                outputPath = buildOutputPath(outputDir, index),
                format = ImageFormat.PNG,
                quality = 100  // 无损
            )
            
            Screenshot(
                id = UUID.randomUUID().toString(),
                type = ScreenshotType.values()[index],
                filePath = screenshot.path,
                timestampMs = timestamp,
                width = frame.width,
                height = frame.height,
                format = ImageFormat.PNG,
                fileSize = screenshot.size
            )
        }.let { Result.success(it) }
    }
}
```

### 6.1 ONNX Runtime 集成

```kotlin
class OnnxInferenceEngine(
    private val modelPath: String,
    private val numThreads: Int = 4
) {
    private var session: OrtSession? = null
    private var environment: OrtEnvironment? = null
    
    fun initialize(): Result<Unit> {
        return try {
            environment = OrtEnvironment.getEnvironment()
            session = environment!!.createSession(
                modelPath,
                OrtSession.SessionOptions().apply {
                    setIntraOpNumThreads(numThreads)
                    setInterOpNumThreads(numThreads)
                    enableMemoryPattern()
                    setExecutionMode(OrtExecutionMode.ORT_SEQUENTIAL)
                }
            )
            Result.success(Unit)
        } catch (e: Exception) {
            Result.failure(e)
        }
    }
    
    suspend fun run(inputs: Map<String, OnnxTensor>): Result<Array<OnnxTensor>> {
        return withContext(Dispatchers.Default) {
            try {
                val outputs = session!!.run(inputs)
                Result.success(outputs.map { it as OnnxTensor }.toTypedArray())
            } catch (e: Exception) {
                Result.failure(e)
            }
        }
    }
    
    fun release() {
        session?.close()
        environment?.close()
    }
}
```

```kotlin
class VehicleDetectorImpl(

    private val config: VehicleDetectorConfig

                color = null

            0 -> VehicleCategory.CAR
            1 -> VehicleCategory.TRUCK
            2 -> VehicleCategory.BUS
            3 -> VehicleCategory.MOTORCYCLE
            else -> VehicleCategory.UNKNOWN
        }
    }
```

```kotlin
class AnalysisViewModel @Inject constructor(

            }
        }
    }
    
    fun cancelAnalysis() {
        currentTaskId?.let { taskId ->
            viewModelScope.launch {
                cancelAnalysisUseCase(taskId)
                currentTaskId = null
            }
        }
    }
    
    override fun onCleared() {
        super.onCleared()
        currentTaskId?.let { taskId ->
            viewModelScope.launch { cancelAnalysisUseCase(taskId) }
        }
    }
```

```kotlin
class FrameCache(
    private val maxSize: Int = 30,
    private val maxAgeMs: Long = 60000
) {
    private val cache = LinkedHashMap<Long, Frame>(maxSize, 0.75f, true)
    
    fun get(timestampMs: Long): Frame? {
        val frame = cache[timestampMs]
        if (frame != null && isValid(timestampMs)) return frame
        cache.remove(timestampMs)
        return null
    }
    
    fun put(timestampMs: Long, frame: Frame) {
        cache[timestampMs] = frame
        evictIfNeeded()
    }
    
    private fun evictIfNeeded() {
        while (cache.size > maxSize) {
            val oldest = cache.keys().next()
            cache.remove(oldest)
        }
    }
    
    fun clear() = cache.clear()
}
```

### 9.3 内存优化 - Bitmap池化

```kotlin
class BitmapPool(private val maxBitmaps: Int = 4) {
    private val pool = ArrayDeque<Bitmap>(maxBitmaps)
    
    fun acquire(width: Int, height: Int, config: Bitmap.Config): Bitmap {
        val iterator = pool.iterator()
        while (iterator.hasNext()) {
            val bitmap = iterator.next()
            if (bitmap.width == width && bitmap.height == height && bitmap.config == config) {
                iterator.remove()
                bitmap.eraseColor(Color.TRANSPARENT)
                return bitmap
            }
        }
        return Bitmap.createBitmap(width, height, config)
    }
    
    fun release(bitmap: Bitmap) {
        if (pool.size < maxBitmaps) {
            pool.add(bitmap)
        } else {
            bitmap.recycle()
        }
    }
    
    fun clear() {
        pool.forEach { it.recycle() }
        pool.clear()
    }
}
```
